namespace Ldap\Exception\Config;

/**
 * Exception to indicate a missing configuration key
 *
 * @package Ldap\Config;
 */
class MissingConfigKeyException
extends InvalidConfigException
{
    /**
     * Create a new exception
     * 
     * @param string $key Missing key
     */
    public function __construct($key)
    {
        parent::__construct("Missing configuration key '$key'.");
    }
}
